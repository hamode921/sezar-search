// وظيفة صيد الحيتان المحدثة لتمييز العملات القوية فقط async function updateWhaleHunt() { try { const res = await fetch('https://api.binance.com/api/v3/ticker/24hr'); const data = await res.json(); const list = document.getElementById('whale-hunt-list'); let candidates = data.filter(c => { const vol = parseFloat(c.quoteVolume); const chg = parseFloat(c.priceChangePercent); // تصفية أولية للعملات المستقرة وبحجم تداول جيد return c.symbol.endsWith('USDT') && chg >= -1.5 && chg <= 3.5 && vol > 10000000; }).map(c => { const vol = parseFloat(c.quoteVolume); const chg = parseFloat(c.priceChangePercent); // معادلة قياس قوة التجميع let score = (vol / 50000000) - Math.abs(chg); return { ...c, huntScore: score }; }).sort((a,b) => b.huntScore - a.huntScore).slice(0, 12); let html = ''; candidates.forEach(c => { // تظهر العلامة فقط إذا تجاوزت قوة التجميع حد معين (مثلاً 5) const isTopAccumulation = c.huntScore > 5; html += `
${c.symbol} ${isTopAccumulation ? 'ACCUMULATION 🔥 🐋' : ''}
PRE-EXPLOSION: +5% ~ +15%
VOL: ${(parseFloat(c.quoteVolume)/1000000).toFixed(1)}M | CHG: ${c.priceChangePercent}%
`; }); list.innerHTML = html; } catch (e) {} }
