// وظيفة صيد الحيتان المحدثة مع التنبيه الجانبي async function updateWhaleHunt() { try { const res = await fetch('https://api.binance.com/api/v3/ticker/24hr'); const data = await res.json(); const list = document.getElementById('whale-hunt-list'); let candidates = data.filter(c => { const change = parseFloat(c.priceChangePercent); const volume = parseFloat(c.quoteVolume); return c.symbol.endsWith('USDT') && change >= -1 && change <= 3 && volume > 10000000; }).map(c => { const vol = parseFloat(c.quoteVolume); const score = (vol / 50000000) + (parseFloat(c.priceChangePercent) * 2); return { ...c, huntScore: score }; }).sort((a,b) => b.huntScore - a.huntScore).slice(0, 10); let html = ''; candidates.forEach(c => { const isStrong = c.huntScore > 5; html += `
${c.symbol} ACCUMULATION 🔥 🐋
PRE-EXPLOSION: +5% ~ +15%
VOL: ${(parseFloat(c.quoteVolume)/1000000).toFixed(1)}M | CHG: ${c.priceChangePercent}%
`; }); list.innerHTML = html; } catch (e) {} }
